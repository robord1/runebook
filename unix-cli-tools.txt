	#!/bin/bash

	echo "Unlocking encrypted Nextcloud data volume..."
	sudo cryptsetup open /mnt/ord_media/ncdata.img ncdata || exit 1

	echo "Mounting decrypted volume..."
	sudo mount /dev/mapper/ncdata /mnt/mydocs/nextcloud_data || exit 1

	echo "Starting Nextcloud..."
	cd ~/nextcloud || exit 1
	docker compose up -d

	echo "Done. Cloud is running with encrypted data."
